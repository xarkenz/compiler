function fibonacci(limit: i32) -> i32 {
    let a: i32 = 0;
    let b: i32 = 1;

    while (b < limit) {
        let temp: i32 = a + b;
        a = b;
        b = temp;
    }

    return a;
}

function my_print_fn(value: i32) {
    print value;
}

function main() -> i32 {
    let my_fib: i32 = fibonacci(1000);
    my_print_fn(my_fib);

    return 0;
}
